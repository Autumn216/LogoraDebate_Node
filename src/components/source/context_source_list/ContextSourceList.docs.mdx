import { ContextSourceList } from './ContextSourceList';
import { IntlProvider } from 'react-intl';

A component that renders a toggle box for sources.

It should be wrapped around [react-intl](https://formatjs.io/docs/getting-started/installation/)'s IntlProvider.

### Component usage
```js
<IntlProvider locale="en">
    <ContextSourceList
        sources={{
            source_url: "#",
            id: 1,
            origin_image_url: "https://picsum.photos/200",
            publisher: "Logora",
            title: "Title of article",
            published_date: "2022-02-25T11:00:00.000Z"
        },
        {
            source_url: "#",
            id: 2,
            origin_image_url: "https://picsum.photos/200",
            publisher: "Logora",
            title: "An another article",
            published_date: "2022-08-11T11:00:00.000Z"
        }}
    />
</IntlProvider>
```

### Properties
| **Name**      | **Type** | **Required** | **Default** | **Description** |
| ----------- | ----------- | :-----------: | ----------- | ----------- |
| <font color='#4646c6'>sources</font> | <font color="#397300">array</font> | &check; | - | <font color="grey">*list of sources*</font> |
