import { TextEditor } from './TextEditor';

A component that renders a input for text edition with rich content options : 
- bold
- italic
- underline
- quote
- ordered list

### Component usage
```js
import TextEditor from "@logora/debate.input.text-editor";
```
```js
<TextEditor 
    handleChange={handleChange}
    placeholder={"Your placeholder"}
    onSubmit={(textContent, richContent, sources) => console.log(textContent)}
    onActivation={() => console.log("Activated !")}
    disabled={false}
/>
```
