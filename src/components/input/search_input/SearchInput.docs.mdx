import { SearchInput } from './SearchInput';
import { ConfigProvider } from '@logora/debate.context.config_provider'

An input to handle search queries with custom placeholder.

### Component usage

```js
import SearchInput from '@logora/debate.input.search_input';
```

```js
<ConfigProvider>
    <SearchInput 
        placeholder="Search"
        onSearchSubmit={(query) => console.log("SearchInput callback")} 
        disabled={false}
    />
</ConfigProvider>
```

### Properties
| **Name**      | **Type** | **Required** | **Default** | **Description** |
| ----------- | ----------- | :-----------: | ----------- | ----------- |
| <font color='#4646c6'>placeholder</font> | <font color="#397300">string</font> | &check; | - | <font color="grey">*input placeholder*</font> |
| <font color='#4646c6'>disabled</font> | <font color="#397300">boolean</font> | &cross; | false | <font color="grey">*whether input is disabled or not*</font> |
| <font color='#4646c6'>onSearchSubmit</font> | <font color="#397300">string</font> | &cross; | - | <font color="grey">*callback called on search submit*</font> |
