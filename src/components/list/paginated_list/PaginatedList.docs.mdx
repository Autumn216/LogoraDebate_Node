import { PaginatedList } from '@logora/debate.list.paginated_list';
import { IntlProvider } from 'react-intl';
import { ListProvider } from '@logora/debate.list.list_provider';

A list component.

It must be wrapped by [react-intl](https://formatjs.io/docs/getting-started/installation/)'s IntlProvider.
It must be wrapped by DataProvider.
It must be wrapped by ListProvider.

### Component usage
```js
import { PaginatedList } from '@logora/debate.list.paginated_list';
```
```js
<IntlProvider locale="en">
    <ListProvider>
        <DataProviderContext.Provider value={{ dataProvider: data }}>
            <PaginatedList 
                currentListId={"listId"}
                resource={"apiPath"}
                loadingComponent={null}
                resourcePropName="res"
                filters={ {"filter": true} }
                display={"column"}
                perPage={4}
                withPagination={false}
                staticContext={null}
                staticResourceName={"null"}
                sort={"-created_at"}
                countless={true}
            />
        </DataProviderContext.Provider>
    <ListProvider>
</IntlProvider>
```
